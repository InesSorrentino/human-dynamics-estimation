cmake_minimum_required(VERSION 2.6)
project(hde-interface)

find_package(YARP)

yarp_idl_to_dir("../human-forces-provider/thrifts/Force6D.thrift" "${CMAKE_CURRENT_BINARY_DIR}/autogenerated" FT_THRIFT_SOURCES FT_THRIFT_HEADERS FT_THRIFT_INCLUDE_DIRS)

set(hde_interface_headers include/hdeinterfacemodule.h include/hdedriver.h)
set(hde_interface_sources src/main.cpp src/hdeinterfacemodule.cpp src/hdedriver.cpp)

include_directories(${YARP_INCLUDE_DIRS} ${FT_THRIFT_INCLUDE_DIRS} -I include/)

add_executable(hde-interface ${hde_interface_headers} ${hde_interface_sources} ${FT_THRIFT_SOURCES} ${FT_THRIFT_HEADERS})

target_link_libraries(hde-interface ${YARP_LIBRARIES})
